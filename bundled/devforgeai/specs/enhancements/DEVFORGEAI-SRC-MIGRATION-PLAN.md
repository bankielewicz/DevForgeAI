## Executive Summary

This specification defines the migration of DevForgeAI from "development-in-operational-folders" to "development-in-src-with-installer-deployment" pattern, enabling the framework to be installed in external projects and tested independently.

**Current State:**
- Development occurs directly in `.claude/` and `devforgeai/` (operational folders)
- Framework cannot be tested in external projects
- No separation between source and deployed files
- Manual installation required (copy .claude/ folder)

**Target State:**
- Source files in `src/claude/` and `src/devforgeai/`
- Installer deploys: src/claude/ → .claude/, src/devforgeai/ → devforgeai/
- Version-aware installation with backup/rollback
- External project testing enabled
- Proper SDLC: source → build → deploy → test

**Scope:**
- ~370 framework files (full framework including feedback system)
- 8-phase migration over 4 sprints (8 weeks)
- 50 story points, 54-83 developer hours

## 2. Installer Architecture

### 2.1 Installation Modes

**Mode 1: Fresh Install**
- Target: Project with no existing .claude/ or devforgeai/
- Action: Full deployment of all framework files
- CLAUDE.md: Merge framework template with existing CLAUDE.md (if present)
- Version: Write version.json to devforgeai/

**Mode 2: Upgrade**
- Target: Project with existing DevForgeAI installation
- Detection: Check for devforgeai/.version.json
- Action: Version comparison, selective file updates
- Backup: Automatic backup before upgrade
- CLAUDE.md: Smart merge (preserve user sections, update framework sections)
- Conflicts: User prompted for resolution

**Mode 3: Rollback**
- Target: Project with failed installation/upgrade
- Detection: Check for .backups/devforgeai-backup-{timestamp}/
- Action: Restore from backup
- Verification: Checksum validation
- CLAUDE.md: Restore from backup

**Mode 4: Validate**
- Target: Any project with DevForgeAI
- Action: Check file integrity, version consistency
- Report: List missing files, version mismatches
- No changes: Read-only validation

**Mode 5: Uninstall**
- Target: Project with DevForgeAI to remove
- Action: Backup then delete .claude/ and devforgeai/ framework sections
- CLAUDE.md: Remove framework sections, preserve user sections
- Cleanup: Remove version.json

### 2.2 Version Detection Logic

```python
def get_installed_version(project_path):
    """Detect installed DevForgeAI version."""
    version_file = Path(project_path) / 'devforgeai/.version.json'

    if not version_file.exists():
        return None  # No installation

    version_data = json.loads(version_file.read_text())
    return {
        'version': version_data['version'],        # e.g., "1.0.0"
        'installed_at': version_data['installed_at'],
        'schema_version': version_data.get('schema_version', '1.0')
    }

def get_source_version(devforgeai_repo):
    """Get version from DevForgeAI source."""
    version_file = Path(devforgeai_repo) / 'src/devforgeai/version.json'

    if not version_file.exists():
        raise InstallerError("version.json not found in source")

    return json.loads(version_file.read_text())

def compare_versions(installed, source):
    """Compare versions and determine update strategy."""
    from packaging import version

    if installed is None:
        return 'fresh_install'

    installed_ver = version.parse(installed['version'])
    source_ver = version.parse(source['version'])

    if source_ver > installed_ver:
        # Determine update type
        if source_ver.major > installed_ver.major:
            return 'major_upgrade'  # Breaking changes
        elif source_ver.minor > installed_ver.minor:
            return 'minor_upgrade'  # New features
        else:
            return 'patch_upgrade'  # Bug fixes
    elif source_ver == installed_ver:
        return 'reinstall'  # Same version
    else:
        return 'downgrade'  # Source older than installed
```

### 2.3 Backup Strategy

```python
def create_backup(project_path, backup_reason='upgrade'):
    """Create timestamped backup before any changes."""
    timestamp = datetime.now().strftime('%Y%m%d-%H%M%S')
    backup_dir = Path(project_path) / f'.backups/devforgeai-{backup_reason}-{timestamp}'

    # Backup operational folders
    if Path(project_path, '.claude').exists():
        shutil.copytree(Path(project_path, '.claude'),
                       backup_dir / '.claude')

    if Path(project_path, 'devforgeai').exists():
        shutil.copytree(Path(project_path, 'devforgeai'),
                       backup_dir / 'devforgeai')

    # Backup CLAUDE.md
    claude_md = Path(project_path, 'CLAUDE.md')
    if claude_md.exists():
        shutil.copy2(claude_md, backup_dir / 'CLAUDE.md')

    # Write backup manifest
    manifest = {
        'created_at': timestamp,
        'reason': backup_reason,
        'files_backed_up': count_files(backup_dir),
        'total_size_mb': get_dir_size_mb(backup_dir),
        'original_version': get_installed_version(project_path)
    }

    (backup_dir / 'manifest.json').write_text(json.dumps(manifest, indent=2))

    return backup_dir

def rollback(project_path, backup_timestamp=None):
    """Restore from backup."""
    backups_dir = Path(project_path) / '.backups'

    if backup_timestamp:
        backup_dir = backups_dir / f'devforgeai-*-{backup_timestamp}'
    else:
        # Use most recent backup
        backups = sorted(backups_dir.glob('devforgeai-*'))
        if not backups:
            raise InstallerError("No backups found")
        backup_dir = backups[-1]

    # Verify backup integrity
    manifest = json.loads((backup_dir / 'manifest.json').read_text())
    if count_files(backup_dir) != manifest['files_backed_up']:
        raise InstallerError("Backup corrupted")

    # Restore files
    if (backup_dir / '.claude').exists():
        shutil.rmtree(Path(project_path, '.claude'))
        shutil.copytree(backup_dir / '.claude',
                       Path(project_path, '.claude'))

    if (backup_dir / 'devforgeai').exists():
        shutil.rmtree(Path(project_path, 'devforgeai'))
        shutil.copytree(backup_dir / 'devforgeai',
                       Path(project_path, 'devforgeai'))

    # Restore CLAUDE.md
    if (backup_dir / 'CLAUDE.md').exists():
        shutil.copy2(backup_dir / 'CLAUDE.md',
                    Path(project_path, 'CLAUDE.md'))

    print(f"✅ Rolled back to: {manifest['created_at']}")
    return True
```

### 2.4 CLAUDE.md Merge Algorithm

```python
def merge_claude_md(source_template, existing_file, variables):
    """
    Intelligent merge of framework CLAUDE.md with user's existing file.

    Strategy:
    1. Parse both files into sections
    2. Substitute variables in template
    3. Detect conflicts
    4. Merge with user approval
    5. Validate result
    """
    # Step 1: Read files
    template = read_file(source_template)  # src/CLAUDE.md
    existing = read_file(existing_file) if Path(existing_file).exists() else ""

    # Step 2: Parse into sections
    template_sections = parse_markdown_sections(template)
    user_sections = parse_markdown_sections(existing) if existing else {}

    # Step 3: Substitute variables
    for section in template_sections.values():
        section.content = substitute_variables(section.content, variables)

    # Step 4: Detect conflicts
    conflicts = []
    for section_name in template_sections:
        if section_name in user_sections:
            # Conflict: Both have same section
            conflicts.append({
                'section': section_name,
                'template': template_sections[section_name],
                'user': user_sections[section_name]
            })

    # Step 5: Build merged document
    merged = build_merged_document(
        template_sections,
        user_sections,
        conflicts,
        strategy='preserve_user_append_framework'
    )

    # Step 6: User review
    show_diff(existing, merged)
    if user_approves():
        return merged
    else:
        return handle_user_modifications(merged)

def substitute_variables(content, variables):
    """Replace template variables with project-specific values."""
    replacements = {
        '{{PROJECT_NAME}}': variables.get('project_name', 'MyProject'),
        '{{PROJECT_PATH}}': variables.get('project_path', os.getcwd()),
        '{{PYTHON_VERSION}}': variables.get('python_version', 'auto-detect'),
        '{{PYTHON_PATH}}': variables.get('python_path', 'auto-detect'),
        '{{TECH_STACK}}': variables.get('tech_stack', 'Not configured'),
        '{{INSTALLATION_DATE}}': datetime.now().strftime('%Y-%m-%d'),
    }

    for placeholder, value in replacements.items():
        content = content.replace(placeholder, str(value))

    return content

def build_merged_document(template_sections, user_sections, conflicts, strategy):
    """
    Build final CLAUDE.md from template + user content.

    Strategy: preserve_user_append_framework
    - User's original content at top
    - Framework sections follow
    - Conflicts: Keep user version, add framework as subsection
    """
    merged = []

    # Add user's original sections first (if any)
    if user_sections:
        merged.append("<!-- USER'S ORIGINAL INSTRUCTIONS -->")
        for name, section in user_sections.items():
            if name not in template_sections:
                # User section, no conflict
                merged.append(section.content)

    # Add framework marker
    merged.append("\n\n---\n\n")
    merged.append("<!-- DEVFORGEAI FRAMEWORK (AUTO-GENERATED) -->")
    merged.append("<!-- Last Updated: {{INSTALLATION_DATE}} -->")
    merged.append("<!-- Version: {{FRAMEWORK_VERSION}} -->")
    merged.append("\n")

    # Add framework sections
    for name, section in template_sections.items():
        if name in user_sections and name in [c['section'] for c in conflicts]:
            # Conflict: Add as subsection
            merged.append(f"\n### {name} (DevForgeAI Framework)\n")
            merged.append(section.content)
        else:
            # No conflict: Add normally
            merged.append(section.content)

    return '\n'.join(merged)
```

### 2.5 File Deployment Logic

```python
def deploy_framework(source_root, target_project, mode='full'):
    """
    Deploy framework files from src/ to target project.

    Args:
        source_root: Path to DevForgeAI repository (contains src/)
        target_project: Path to target project (receives .claude/)
        mode: 'full', 'selective', 'dry-run'
    """
    source_claude = Path(source_root) / 'src/claude'
    source_devforgeai = Path(source_root) / 'src/devforgeai'

    target_claude = Path(target_project) / '.claude'
    target_devforgeai = Path(target_project) / 'devforgeai'

    # Backup if existing installation
    if target_claude.exists() or target_devforgeai.exists():
        backup_dir = create_backup(target_project, reason='upgrade')
        print(f"✅ Backup created: {backup_dir}")

    # Deploy .claude/
    if mode == 'dry-run':
        preview_changes(source_claude, target_claude)
    else:
        deploy_tree(source_claude, target_claude,
                   exclude=['*.backup*', '__pycache__', '*.pyc',
                           '.egg-info', 'htmlcov', '.coverage'])
        print(f"✅ Deployed: {count_files(target_claude)} files to .claude/")

    # Deploy devforgeai/
    if mode == 'dry-run':
        preview_changes(source_devforgeai, target_devforgeai)
    else:
        deploy_tree(source_devforgeai, target_devforgeai,
                   exclude=['qa/reports', 'RCA', 'adrs', 'feedback/imported',
                           'logs', 'stories'])
        print(f"✅ Deployed: {count_files(target_devforgeai)} files to devforgeai/")

    # Merge CLAUDE.md
    source_template = Path(source_root) / 'src/CLAUDE.md'
    target_claude_md = Path(target_project) / 'CLAUDE.md'

    variables = detect_project_variables(target_project)
    merged_content = merge_claude_md(source_template, target_claude_md, variables)

    if mode != 'dry-run':
        target_claude_md.write_text(merged_content)
        print("✅ CLAUDE.md merged")

    # Install CLI
    if mode != 'dry-run':
        install_cli(target_project)

    # Write version tracking
    if mode != 'dry-run':
        write_version_file(target_project, get_source_version(source_root))

    return True

def deploy_tree(src, dst, exclude=None):
    """Copy directory tree with exclusions."""
    exclude = exclude or []

    for item in src.rglob('*'):
        # Skip excluded patterns
        if any(pattern in str(item) for pattern in exclude):
            continue

        relative_path = item.relative_to(src)
        target_path = dst / relative_path

        if item.is_dir():
            target_path.mkdir(parents=True, exist_ok=True)
        else:
            target_path.parent.mkdir(parents=True, exist_ok=True)
            shutil.copy2(item, target_path)

            # Set executable permissions for scripts
            if item.suffix in ['.sh', '.py'] and item.name in EXECUTABLE_SCRIPTS:
                target_path.chmod(0o755)
```

```python
def selective_update(source_root, target_project, installed_version, source_version):
    """Update only changed files (for minor/patch upgrades)."""
    changes = compute_file_changes(source_root, target_project,
                                   installed_version, source_version)

    print(f"Files to update: {len(changes['modified'])}")
    print(f"Files to add: {len(changes['added'])}")
    print(f"Files to remove: {len(changes['removed'])}")

    # User confirmation
    if not user_confirms_changes(changes):
        return False

    # Backup before changes
    backup_dir = create_backup(target_project, reason='selective-update')

    # Apply changes
    for file_path in changes['modified'] + changes['added']:
        source_file = Path(source_root) / 'src' / file_path
        target_file = Path(target_project) / file_path

        # Skip user configs
        if is_user_config(file_path):
            print(f"⏩ Skipped (user config): {file_path}")
            continue

        target_file.parent.mkdir(parents=True, exist_ok=True)
        shutil.copy2(source_file, target_file)

    # Remove obsolete files
    for file_path in changes['removed']:
        target_file = Path(target_project) / file_path
        if target_file.exists():
            target_file.unlink()

    # Update version
    write_version_file(target_project, source_version)

    return True

def compute_file_changes(source_root, target_project, old_version, new_version):
    """Compute what files changed between versions."""
    # This would ideally use git diff between version tags
    # For MVP, compare file timestamps and checksums

    source_files = get_framework_files(Path(source_root) / 'src')
    target_files = get_framework_files(target_project)

    changes = {
        'added': [],
        'modified': [],
        'removed': [],
        'unchanged': []
    }

    for file_path in source_files:
        source_file = Path(source_root) / 'src' / file_path
        target_file = Path(target_project) / file_path

        if not target_file.exists():
            changes['added'].append(file_path)
        elif files_differ(source_file, target_file):
            changes['modified'].append(file_path)
        else:
            changes['unchanged'].append(file_path)

    for file_path in target_files:
        if file_path not in source_files:
            changes['removed'].append(file_path)

    return changes
```
